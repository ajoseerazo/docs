---
title: "Create Payout"
api: "POST https://stablecoin-api.getmeru.com/payouts"
description: "This endpoint allows to initiate a payout process"
---

### Body

<ParamField body="payout" type="array" required>
  This array defines the list of payouts to process

  <Expandable title="Array of objects">
    <ParamField body="email" type="string" required>
      User's email to send the money
    </ParamField>
    <ParamField body="amount" type="number" required>
      Amount to send
    </ParamField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="id" type="boolean">
  The id of the payout process. Use this id to check the status of the
  process
</ResponseField>
