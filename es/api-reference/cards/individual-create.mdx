---
title: "Crear solicitud individual"
api: "POST https://stablecoin-api.sandbox.getmeru.com/api/v1/cards-program/applications/individual"
description: "Este endpoint crea una solicitud de tarjeta individual completa con todos los datos del cliente"
---

### Cuerpo

<ParamField body="firstName" type="string" required>
  Nombre del solicitante (máximo 50 caracteres)
</ParamField>

<ParamField body="lastName" type="string" required>
  Apellido del solicitante (máximo 50 caracteres)
</ParamField>

<ParamField body="birthDate" type="string" required>
  Fecha de nacimiento en formato YYYY-MM-DD
</ParamField>

<ParamField body="nationalId" type="string" required>
  Número de identificación nacional
</ParamField>

<ParamField body="countryOfIssue" type="string" required>
  Código de país de dos letras donde se emitió el documento (por ejemplo, "US")
</ParamField>

<ParamField body="email" type="string" required>
  Correo electrónico del solicitante
</ParamField>

<ParamField body="phoneCountryCode" type="string" required>
  Código de país del teléfono (1-3 caracteres, por ejemplo, "1" para US)
</ParamField>

<ParamField body="phoneNumber" type="string" required>
  Número de teléfono (1-15 caracteres)
</ParamField>

<ParamField body="walletAddress" type="string">
  Dirección de wallet blockchain (opcional)
</ParamField>

<ParamField body="ipAddress" type="string" required>
  Dirección IP del solicitante
</ParamField>

<ParamField body="occupation" type="string" required>
  Ocupación del solicitante
</ParamField>

<ParamField body="annualSalary" type="number" required>
  Salario anual
</ParamField>

<ParamField body="accountPurpose" type="string" required>
  Propósito de apertura de la cuenta
</ParamField>

<ParamField body="expectedMonthlyVolume" type="number" required>
  Volumen mensual de transacciones esperado
</ParamField>

<ParamField body="isTermsOfServiceAccepted" type="boolean" required>
  Si el solicitante aceptó los términos del servicio
</ParamField>

<ParamField body="hasExistingDocuments" type="boolean">
  Si el solicitante ya tiene documentos en archivo
</ParamField>

<ParamField body="address" type="object" required>
  Dirección física del solicitante

  <Expandable title="Objeto address">
    <ParamField body="line1" type="string" required>
      Dirección línea 1 (máximo 100 caracteres)
    </ParamField>
    <ParamField body="line2" type="string">
      Dirección línea 2 (máximo 100 caracteres)
    </ParamField>
    <ParamField body="city" type="string" required>
      Ciudad (máximo 50 caracteres)
    </ParamField>
    <ParamField body="region" type="string" required>
      Estado o región (máximo 50 caracteres)
    </ParamField>
    <ParamField body="postalCode" type="string" required>
      Código postal (1-9 caracteres)
    </ParamField>
    <ParamField body="countryCode" type="string" required>
      Código de país de dos letras (por ejemplo, "US")
    </ParamField>
    <ParamField body="country" type="string">
      Nombre del país (máximo 50 caracteres)
    </ParamField>
  </Expandable>
</ParamField>

### Respuesta

<ResponseField name="success" type="boolean">
  Indica si la operación fue exitosa
</ResponseField>

<ResponseField name="userId" type="string">
  El identificador único del usuario de la solicitud creada
</ResponseField>

<ResponseField name="status" type="string">
  Estado de la solicitud: `pending`, `approved`, `rejected` o `requires_documents`
</ResponseField>
